    <script>
        const CONTENT = {
            models: [
                { id: 'sub', title: 'Power of Subtraction', body: 'Innovation is found in what you remove. Pruning 150+ legacy systems unlocked $275M in value—proving subtraction is the key to scale.', meta: '→ Principle: Optimize via rationalization' },
                { id: 'dis', title: 'Passionate Dispassion', body: 'Loving a project intensely while maintaining the mathematical humility to kill it if the techno-economics don’t close.', meta: '→ Principle: Dispassionate assessment' }
            ],
            learning: [
                { id: 'l1', title: 'Techno-Economic Modeling', body: 'Currently stress-testing the bridge between "Science Fiction" prototypes and market viability at X.', models: ['sub', 'dis'], meta: '→ Status: Active Experiment' },
                { id: 'l2', title: 'Full-Stack Operationalizing', body: 'Architecting Snowflake data lifecycles and Python-based ETL to de-risk non-technical hurdles.', models: ['sub'], meta: '→ Skill: Technical Foundation' }
            ],
            retrospect: [
                { id: 'r1', title: '$275M Portfolio Rationalization', body: 'Redesigning high-volume operational workflows to achieve a 47% reduction in processing time.', models: ['sub'], meta: '→ Result: 10x Impact' }
            ]
        };

        let canvas, ctx, particles = [], activePos = null;

        function init() {
            canvas = document.getElementById('neuralCanvas');
            ctx = canvas.getContext('2d');
            window.addEventListener('resize', () => { canvas.width = window.innerWidth; canvas.height = window.innerHeight; });
            canvas.width = window.innerWidth; canvas.height = window.innerHeight;

            for(let i=0; i<100; i++) particles.push({ x: Math.random()*canvas.width, y: Math.random()*canvas.height, vx: (Math.random()-0.5)*0.5, vy: (Math.random()-0.5)*0.5 });
            render('learning');
            animate();
        }

        function render(tab) {
            const grid = document.getElementById('contentGrid');
            grid.innerHTML = '';
            CONTENT[tab].forEach(item => {
                const div = document.createElement('div');
                div.className = 'card';
                div.innerHTML = `<div class="card-title">${item.title}</div><div class="card-body">${item.body}</div><div class="card-meta">${item.meta}</div>`;
                div.onclick = (e) => {
                    document.querySelectorAll('.card').forEach(c => c.classList.remove('highlighted'));
                    div.classList.add('highlighted');
                    const rect = div.getBoundingClientRect();
                    activePos = { x: rect.left + rect.width/2, y: rect.top + rect.height/2 };
                };
                grid.appendChild(div);
            });
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            particles.forEach(p => {
                let speed = 1;
                if(activePos) {
                    const d = Math.sqrt((p.x-activePos.x)**2 + (p.y-activePos.y)**2);
                    if(d < 300) speed = 5 * (1 - d/300);
                }
                p.x += p.vx * speed; p.y += p.vy * speed;
                if(p.x<0 || p.x>canvas.width) p.vx *= -1;
                if(p.y<0 || p.y>canvas.height) p.vy *= -1;
                
                ctx.fillStyle = activePos && Math.sqrt((p.x-activePos.x)**2 + (p.y-activePos.y)**2) < 300 ? 'rgba(155, 89, 182, 0.8)' : 'rgba(255,255,255,0.2)';
                ctx.beginPath(); ctx.arc(p.x, p.y, 2, 0, Math.PI*2); ctx.fill();
            });
            requestAnimationFrame(animate);
        }

        document.querySelectorAll('.nav-tab').forEach(t => t.onclick = () => {
            document.querySelectorAll('.nav-tab').forEach(nt => nt.classList.remove('active'));
            t.classList.add('active');
            render(t.dataset.tab);
            activePos = null;
        });

        init();
    </script>
